  + ID [PK]
  --
  Name
  Email
  Phone
}

  + ID [PK]
  --
  Date
  Total
}

  + ID [PK]
  --
  Name
}

  + ID [PK]
  --
  Details
}

entity1 -- entity2 : one-to-many
